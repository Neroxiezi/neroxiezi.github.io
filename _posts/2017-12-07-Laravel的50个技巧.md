                return $model->validate();
            }
        });
    }

    public function validate() { }
}

 {
     public $incrementing = false;
     protected static function boot()
     {
         parent::boot();
         static::creating(function ($model) {
             $key = $model->getKeyName();
             if(empty($model->{$key})) {
                 $model->{$key} = (string)$model->generateNewId();

         });
